User model

enum Role {
    STUDENT, TUTOR
}

User {
    userId: String,
    firstName: String,
    lastName: String,
    email: String,
    password: String,
    role: enum
    createdAt: Timestamp,
    updatedAt: Timestamp,
    profileImageUrl: String
}

Tutor {
    tutorId: String,
    available: boolean,
    createdAt: Timestamp,
    updatedAt: Timestamp,
}

TutorSubject {
    id: String,
    tutorId: String (FK),
    subjectCode: String (FK),
    createdAt: Timestamp,
}

Subject {
    subjectCode: String,
    subjectName: String
}

Review {
    reviewId: String,
    comment: String,
    rating: number,
    studentId: String (FK),
    tutorId: String (FK)
}

ReviewLikes {
    rLikesId: String,
    studentId: String (FK),
    reviewId: String (FK),
}

Student {
    studentId: String,
}
